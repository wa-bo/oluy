```bash
> apt update 
> pkg upgrade 
> pkg install git -y
> pkg install nodejs -y
> pkg install ffmpeg -y
> pkg install imagemagick -y
> pkg install yarn
> git clone https://github.com/wa-bo/oleg-bot.git
> cd jok
> yarn install 
> npm update
> npm start
```
